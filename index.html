Variables

example


//CSS
h1 { color: #C45655; }
h2 { color: #C45655; }
h3 { color: #C45655; }
h4 { color: #C45655; }


						

//SASS
$fuzzyWuzzyBrown: #C45655;

h1 { color: $fuzzyWuzzyBrown; }
h2 { color: $fuzzyWuzzyBrown; }
h3 { color: $fuzzyWuzzyBrown; }
h4 { color: $fuzzyWuzzyBrown; }

Nesting


//CSS
.myDiv {
    background:black;
}

.myDiv ul {
    list-style-type: square;
}

.myDiv ul li {
    color:green;
}
						

//SASS
.myDiv {
    background: black;

    ul {
        list-style-type: square;

        li {
            color:green;
        }
    }
}
